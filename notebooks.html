# Menghitung jarak data Nominal/categorical
def d1(i,j):
    P = 4
    M = 0
    for col in ["Mobile_Phone", "Model", "Review","Payment_Method"]:
        if mobile[col][i] == mobile[col][j]:
            M += 1

    return (P-M)/P

# Menampilkan hasil perhitungan jarak di tipe data nominal/categorical
print(d1(1,2))
print(d1(1,3))
print(d1(1,4))

def d2(i,j,h):
    result = 0
    for col in [hasil1, hasil2, hasil3, hasil4, hasil5]:
        result = math.pow(abs(col[i] - col[j]), h)
    return result**(1/h)
