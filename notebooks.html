# Menghitung jarak data Nominal/categorical
def d1(i,j):
    P = 4
    M = 0
    for col in ["Mobile_Phone", "Model", "Review","Payment_Method"]:
        if mobile[col][i] == mobile[col][j]:
            M += 1

    return (P-M)/P

# Menampilkan hasil perhitungan
print(d1(1,2))
print(d1(1,3))
print(d1(1,4))

# Menghitung jarak data Numeric dengan cara Euclidean
def d2(i,j,h):
    result = 0
    for col in ["RAM_GB", "ROM_GB", "Latest_Price_Taka", "Discount", "Selling_Price"]:
        result = math.pow(abs(mobile[col][i] - mobile[col][j]), h)
    return result**(1/h)

# Menampilkan hasil perhitungan
print(d2(1,2,2))
print(d2(1,3,2))
print(d2(1,4,2))
